SELECT avg(waga), COUNT(*) FROM kreatura GROUP BY rodzaj;
SELECT avg(YEAR(CURDATE())-YEAR(dataUR)) AS wiek FROM kreatura GROUP BY rodzaj;

SELECT COUNT(DISTINCT nazwa) FROM zasob GROUP BY rodzaj HAVING MIN(ilosc)>1;
